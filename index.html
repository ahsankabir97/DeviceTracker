var SHEET = 'data';

function doPost(e){
  var lock = LockService.getScriptLock();
  lock.tryLock(15000);
  try{
    var raw = e.postData ? e.postData.contents : '';
    if(!raw) return out(JSON.stringify({ok:false,err:'no data'}));
    // Validate it's JSON
    JSON.parse(raw); // throws if invalid
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sh = ss.getSheetByName(SHEET) || ss.insertSheet(SHEET);
    sh.clearContents();
    sh.getRange(1,1).setValue(raw); // store full payload as-is
    var ts = new Date().toISOString();
    sh.getRange(1,2).setValue(ts);
    return out(JSON.stringify({ok:true, ts:ts}));
  }catch(err){
    Logger.log(err.toString());
    return out(JSON.stringify({ok:false, err:err.toString()}));
  }finally{ lock.releaseLock(); }
}

function doGet(e){
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sh = ss.getSheetByName(SHEET);
  var val = sh ? sh.getRange(1,1).getValue() : '';
  var ts  = sh ? String(sh.getRange(1,2).getValue()) : '';
  // Return {data: "...json string...", ts: "..."}
  return out(JSON.stringify({data: val||'{}', ts: ts}));
}

function out(s){
  return ContentService.createTextOutput(s)
    .setMimeType(ContentService.MimeType.JSON);
}
